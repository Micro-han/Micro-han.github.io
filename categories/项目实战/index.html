Unity初见之GoogleCardBoardVR小项目.md

前言

  • 网上的Unity博客介绍实在是太乱太复杂了，正好适逢我虚拟现实课要写一个基于Unity和GoogleCardBoard的VR项目，很烦，赶紧记录一下。

环境配置

  • Unity2021版本移除了XR Settings，而是集成到了一个插件之中，但是我很傻瓜，我并不会配置这个XR Settings。所以VR开发目前的建议是选择Unity2019版本。

  • 环境配置具体参考https://developers.google.com/cardboard/develop/unity/quickstart官方文档，那我来翻译一下。

  • 1、在Unity中，打开Window → Package Manage

    2、选择Add package from git URL，粘贴https://github.com/googlevr/cardboard-xr-plugin.git

    3、并且正常导入一些所需要的包比如GoogleVR等显而易见的包，去File → Building Settings，点击Android选择Switch Platform

    4、打开Player Settings，

    ​ - Project Settings > Player > Resolution and Presentation.

    ​ 将 Default Orientation to Landscape Left or Landscape Right.

    ​ 禁用 Optimized Frame Pacing.

    ​ - Navigate to Project Settings > Player > Other Settings.

    ​ Choose OpenGLES2, or OpenGLES3, or both in Graphics APIs.

    ​ Select IL2CPP in Scripting Backend.

    ​ Select desired architectures by choosing ARMv7, ARM64, or both in Target Architectures.

    ​ Select Require in Internet Access.

    ​ Specify your company domain under Package Name.

    5、Project Settings > Player > Publishing Settings。选择 Custom Main Gradle Template和Gradle Properties Template。Assets/Plugins/Android/mainTemplate.gradle中加入代码

    6、在Assets/Plugins/Android/gradleTemplate.properties中加入代码

    7、设置SDK最低版本为24，最高版本为30。同时，选择Custom Main Manifest，在Assets/Plugins/Android/AndroidManifest.xml中加入代码

    8、在XR plugin Managment中选择CardBoard XR Plugin

    9、XR Settings里选择Virtual Reality Settings支持，并加入GoogleCardboard支持。

using UnityEngine;

namespace SwordWorld
{
    public class player_movement
        : MonoBehaviour
    {
        public float walk_speed = 6f;
        public float run_speed = 12f;


        private Vector3 movement;
        private Animator animator;
        private Rigidbody playerRigidbody;

        // rotate
        public float turnSmoothing = 3.0f;
        private Transform cameraTransform;
        private bool isWalk;
        private bool isRun;
        private bool cnt;
        private float h;
        private float v;
        private float num = 1;
        
        void Awake()
        {
            // Set up references.
            animator = GetComponent<Animator>();
            playerRigidbody = GetComponent<Rigidbody>();

            cameraTransform = Camera.main.transform;
        }

        void Update()
        {
            if(num <= 200) {
                h = 0;
                v = -1;
                num = num + 1;
            } else {
                h = 0;
                v = 1;
                num = num + 1;
                if(num == 400) num = 0;
            }
            isWalk = true;
        }

       

        void FixedUpdate()
        {
            // Move the player around the scene.
            Move(h, v);

            // Turn the player to face the mouse cursor. 
            Rotate(h, v);

            // Jump
            Jump(h, v);
        }

        void Move(float h, float v)
        {
            float speed = isRun ? run_speed : walk_speed;

            // Set the movement vector based on the axis input.
            movement.Set(h, 0.0f, v);

            // Normalise the movement vector and make it proportional to the speed per second.
            movement = movement.normalized * speed * Time.deltaTime;
            
            // Move the player to it's current position plus the movement.
            playerRigidbody.MovePosition(transform.position + movement);

            // Animator
            {
                if (isRun)
                {
                    animator.SetBool("IsRun", isRun);
                }
                else
                {
                    animator.SetBool("IsRun", isRun);
                    animator.SetBool("IsWalk", isWalk);
                }
            }
        }

        void Jump(float h, float v)
        {
            if (isJump)
            {
                animator.SetTrigger("Jump");
                playerRigidbody.velocity = new Vector3(0, jumpHeight, 0);
            }
        }

        Vector3 Rotate(float h, float v)
        {
            Vector3 forward = cameraTransform.TransformDirection(Vector3.forward);
            forward = forward.normalized;

            Vector3 right = new Vector3(forward.z, 0, -forward.x);

            Vector3 targetDirection;
            targetDirection = forward * v + right * h;

            if ((isWalk && targetDirection != Vector3.zero))
            {
                Quaternion targetRotation = Quaternion.LookRotation(targetDirection, Vector3.up);

                Quaternion newRotation = Quaternion.Slerp(GetComponent<Rigidbody>().rotation, targetRotation, turnSmoothing * Time.deltaTime);

                // TODO：不知为毛，Rigid 的约束不起作用，只能手动设置为 0 
                newRotation.x = 0f;
                newRotation.z = 0f;
                GetComponent<Rigidbody>().MoveRotation(newRotation);
            }

            return targetDirection;
        }
    }
}
